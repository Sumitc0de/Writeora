  Link,

  Video,
  Wand2,
  Sparkles,
  FilePlus,
  Upload,
} from "lucide-react";

const BlogEditor = () => {
  const [title, setTitle] = useState("");
  const [subtitle, setSubtitle] = useState("");
  const [category, setCategory] = useState("");
  const [headerImage, setHeaderImage] = useState("");
  const [content, setContent] = useState("");
  const [aiLoading, setAiLoading] = useState(false);
  const [template, setTemplate] = useState("");
  const editorRef = useRef(null);

  // Prevent cursor jump on Enter key
  useEffect(() => {
    const editor = editorRef.current;
    if (editor) {
      const handleEnter = (e) => {
        if (e.key === "Enter") {
          document.execCommand("insertHTML", false, "<br><br>");
          e.preventDefault();
        }
      };
      editor.addEventListener("keydown", handleEnter);
      return () => editor.removeEventListener("keydown", handleEnter);
    }
  }, []);

  // Toolbar actions
  const applyStyle = (command, value = null) => {
    document.execCommand(command, false, value);
  };

  const handleImageUpload = () => {
    const url = prompt("Enter Image URL:");
    if (url) document.execCommand("insertImage", false, url);
  };

  const handleVideoInsert = () => {
    const url = prompt("Enter YouTube or video embed URL:");
    if (url)
      document.execCommand(
        "insertHTML",
        false,
        `<iframe src="${url}" width="400" height="250"></iframe>`
      );
  };

  // AI actions simulation
  const handleAIEnhance = (type) => {
    setAiLoading(true);

      `,
    },
    learning: {
      title: "Learn JavaScript with Real-World Projects",
      subtitle: "The 2025 roadmap for mastering frontend development.",
      category: "Learning / Programming",

        <p>JavaScript remains the most powerful language for the web. In 2025, focus on project-based learning to grow faster.</p>

          <li>Build mini projects like Todo App & Weather App.</li>
          <li>Use frameworks like React for scalable UI.</li>
        </ul>
      `,

    knowledge: {
      title: "AI Ethics in a Hyper-Automated World",
      subtitle: "Balancing innovation with responsibility.",
      category: "Ethics / Technology",
      content: `
        <p>As automation grows, ethical AI is the need of the hour. Transparency, fairness, and accountability must guide innovation.</p>
        <p>This article explores how companies can adopt responsible AI strategies in 2025.</p>
      `,

    setTemplate(type);

    setTitle(temp.title);
    setSubtitle(temp.subtitle);
    setCategory(temp.category);
    setContent(temp.content);
  };

            <button
              key={key}
